# src/models/CMakeLists.txt
cmake_minimum_required(VERSION 3.16)

set(MODELS_HEADERS
    include/device_info.h
    include/measurement_data.h
    include/sensor_data.h
    include/system_config.h
)

set(MODELS_SOURCES
    src/device_info.cpp
    src/measurement_data.cpp
    src/sensor_data.cpp
    src/system_config.cpp
)

add_library(models_lib STATIC
    ${MODELS_SOURCES}
    ${MODELS_HEADERS}
)

target_include_directories(models_lib PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_link_libraries(models_lib PUBLIC
    Qt6::Core
)

set_target_properties(models_lib PROPERTIES AUTOMOC ON)
