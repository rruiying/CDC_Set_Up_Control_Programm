# src/models/CMakeLists.txt
cmake_minimum_required(VERSION 3.16)

# 创建models库
add_library(models STATIC
    src/sensor_data.cpp
    src/measurement_data.cpp
    src/system_config.cpp
    src/device_info.cpp
)

target_include_directories(models PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

# 设置C++标准
target_compile_features(models PUBLIC cxx_std_17)

# 如果是Windows平台，定义M_PI
if(WIN32)
    target_compile_definitions(models PRIVATE _USE_MATH_DEFINES)
endif()
